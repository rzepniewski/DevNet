    Network Services Example¶

    Deploy complete network infrastructure with DNS, DHCP, NTP, TFTP, HTTP, TACACS+, LDAP, and SNMP/LibreNMS monitoring services.

    Overview¶

    This example shows how to set up a complete network services infrastructure including all eight supported Docker-based services.

    Configuration¶

    testbed:
  name: "network-services-lab"

  compute:
    vms:
      - name: "tbac_vm01"
        deploy_from: "ova"
        ova_image_name: "tbac-base-ubuntu"
        services:
          dns:
            enable: true
          dhcp:
            enable: true
          ntp:
            enable: true
          tftp:
            enable: true
          http:
            enable: true
          tacacs:
            enable: true
          ldap:
            enable: true
          snmp:
            enable: true

  network_services:
    dns:
      image_name: "mvance/unbound"
      image_version: "1.21.1"
      upstream_servers: ["8.8.8.8", "8.8.4.4"]
      a_records:
        - name: "web.local."
          type: "A"
          value: "192.168.1.10"

    dhcp:
      image_name: "kosdk/dhcpd"
      image_version: "4.4.1-2.3"
      interfaces: ["ens192"]
      subnets:
        - cidr: "192.168.1.0/24"
          range_start: "192.168.1.100"
          range_end: "192.168.1.200"
          router: "192.168.1.1"

    ntp:
      image_name: "cturra/ntp"
      image_version: "latest"
      version: 4
      upstream_servers:
        - "0.pool.ntp.org"

    tftp:
      image_name: "pghalliday/tftp"
      image_version: "latest"
      tftp_docker_root: "/var/tftpboot"
      tftp_boot_dir: "/tftpboot"
      tftp_address: "0.0.0.0"
      tftp_port: 69
      max_block_size: 1468
      timeout: 1
      retransmit: 0
      create_mode: true
      secure_mode: true
      verbosity: 1
      permissive_mode: false

    http:
      image_name: "caddy"
      image_version: "alpine"
      http_port: 80
      https_port: 443
      root_dir: "http"
      sites:
        - name: "files"
          domain: "files.lab.local"
          root: "shared"
          browse: true

    tacacs:
      image_name: "christianbecker/tac_plus-ng"
      image_version: "latest"
      hosts:
        - name: "router01"
          address: "192.168.1.1"
          secret: !vault |
            $ANSIBLE_VAULT;1.1;AES256
            ...
      local_users:
        - name: "tbac"
          password: !vault |
            $ANSIBLE_VAULT;1.1;AES256
            ...
          member: "full_access"
      profiles:
        - name: "full_access"
          attributes:
            shell:
              priv_level: 15

    ldap:
      image_name: "bitnami/openldap"
      image_version: "2.6"
      base_dn: "dc=lab,dc=local"
      admin_password: !vault |
        $ANSIBLE_VAULT;1.1;AES256
        ...
      seed_ous:
        - People
        - Groups
      seed_groups:
        - cn: ldapusers
          gid_number: 5000
      seed_users:
        - uid: tbac-admin
          cn: "TBaC Admin"
          sn: "Admin"
          uid_number: 10001
          gid_number: 5000
          home: /home/tbac-admin
          shell: /bin/bash
          password: !vault |
            $ANSIBLE_VAULT;1.1;AES256
            ...

    snmp:
      service_name: "tbac_librenms"
      image_name: "librenms/librenms"
      image_version: "26.2.0"
      db_password: "secure_db_password"      # Change in production (use ansible-vault)
      db_root_password: "secure_root_pwd"    # Change in production (use ansible-vault)
      web_password: "admin_password"         # Web UI login (use ansible-vault)
      snmp_community: "private"
      timezone: "UTC"
      web_port: 8000
      poller_frequency: 300                  # Alert evaluation interval in seconds (default 300)
      fast_ping: true                        # Detect device down in 60s
      fast_ping_interval: 60
      alert_default_only: false          # false = use per-rule transports list
      alert_transports:
        - name: "Email Ops Team"
          type: "mail"
          is_default: true
          config:
            email: "ops@example.com"
            email_backend: smtp
            email_smtp_host: smtp.example.com
            email_smtp_port: 25
            email_from: '"LibreNMS" <librenms@example.com>'
      alert_rules:
        - name: "Port Down"
          severity: critical
          conditions:
            - field: ports.ifOperStatus
              operator: "="
              value: "down"
            - field: ports.ifAdminStatus
              operator: "="
              value: "up"
          transports: ["Email Ops Team"]
        - name: "Device Down"
          severity: critical
          conditions:
            - field: devices.status
              operator: "="
              value: "0"
          transports: ["Email Ops Team"]
      monitored_devices:
        - hostname: "192.168.1.1"
          community: "private"
          snmp_version: "v2c"
          display_name: "Router1"
        - hostname: "192.168.1.2"
          snmp_version: "v3"
          snmpv3_authname: "snmpv3user"
          snmpv3_authpass: "authpassword123"    # Use ansible-vault in production
          snmpv3_authalgo: "SHA"
          snmpv3_cryptopass: "privpassword456"  # Use ansible-vault in production
          snmpv3_cryptoalgo: "AES"
          display_name: "Switch1"
          # Auth level (authPriv) is auto-detected from provided credentials

    Alert Rule Behavior¶

    By default, TBaC configures alert rules for one-time notification:

    Setting Default Behavior
    count 1 Alert fires once per event
    interval 0m No re-alerting
    recovery true Sends recovery email when device/port comes back up

    This means: one email when down → one email when recovered.

    To override (e.g. re-alert every 15 minutes up to 5 times):

    alert_rules:
  - name: "Device Down"
    severity: critical
    count: 5           # Re-alert up to 5 times
    interval: "15m"    # Every 15 minutes
    recovery: true
    conditions:
      - field: devices.status
        operator: "="
        value: "0"
    transports: ["Email Ops Team"]

    Transport Examples¶

    Slack Transport¶

    network_services:
  snmp:
    alert_default_only: false
    alert_transports:
      - name: "Slack NOC"
        type: "slack"
        config:
          url: "https://hooks.slack.com/services/T00000/B00000/XXXXXXX"
          channel: "#network-alerts"   # Optional: override webhook default channel
    alert_rules:
      - name: "Device Down"
        severity: critical
        conditions:
          - field: devices.status
            operator: "="
            value: "0"
        transports: ["Slack NOC"]

    Microsoft Teams Transport¶

    network_services:
  snmp:
    alert_default_only: false
    alert_transports:
      - name: "Teams NOC Channel"
        type: "microsoft_teams"
        config:
          url: "https://outlook.office.com/webhook/YOUR-WEBHOOK-URL"
          use-json: false              # Set true for JSON MessageCard format
    alert_rules:
      - name: "Port Down"
        severity: critical
        conditions:
          - field: ports.ifOperStatus
            operator: "="
            value: "down"
          - field: ports.ifAdminStatus
            operator: "="
            value: "up"
        transports: ["Teams NOC Channel"]

    Cisco Webex Teams Transport¶

    network_services:
  snmp:
    alert_default_only: true              # true = only send via is_default transports
    alert_transports:
      - name: "Webex NOC Space"
        type: "cisco_webex_teams"
        is_default: true               # Apply to ALL alert rules
        config:
          api-token: "NjE2YzY5..."     # Bot access token from developer.webex.com
          room-id: "Y2lzY29zcGFyay..." # Room ID from GET /v1/rooms
          use-markdown: false           # Enable Markdown formatting
    alert_rules:
      - name: "Device Down"
        severity: critical
        conditions:
          - field: devices.status
            operator: "="
            value: "0"
        # No transports needed — Webex has is_default: true

    Webhook (API) Transport¶

    network_services:
  snmp:
    alert_default_only: false
    alert_transports:
      - name: "PagerDuty"
        type: "webhook"
        config:
          url: "https://events.pagerduty.com/generic/2010-04-15/create_event.json"
          method: "POST"               # GET, POST, or PUT (default: POST)
    alert_rules:
      - name: "Device Down"
        severity: critical
        conditions:
          - field: devices.status
            operator: "="
            value: "0"
        transports: ["PagerDuty"]

    Multiple Transports (Email + Slack + Webex)¶

    network_services:
  snmp:
    alert_default_only: false
    alert_transports:
      - name: "Email Ops"
        type: "mail"
        is_default: true
        config:
          email: "ops@example.com"
          email_backend: smtp
          email_smtp_host: smtp.example.com
      - name: "Slack NOC"
        type: "slack"
        config:
          url: "https://hooks.slack.com/services/T00000/B00000/XXXXXXX"
      - name: "Webex NOC"
        type: "cisco_webex_teams"
        config:
          api-token: "NjE2YzY5..."
          room-id: "Y2lzY29zcGFyay..."
    alert_rules:
      - name: "Port Down"
        severity: critical
        conditions:
          - field: ports.ifOperStatus
            operator: "="
            value: "down"
          - field: ports.ifAdminStatus
            operator: "="
            value: "up"
        transports: ["Slack NOC", "Webex NOC"]
        # Email Ops has is_default: true, so it receives ALL alerts automatically
