In each line there can be more than one value of time, separated by ",". Each single value is in format: h:m:s, where letters means: h - how many hours, m - how many minutes and s - how many seconds. For each line calculate the average value and add the result as additional column.

import pandas as pd

# Load the CSV file
df = pd.read_csv('data2.csv')

# Function to calculate average time
def average_time(time_str):
    times = time_str.split(',')
    total_seconds = sum(int(h) * 3600 + int(m) * 60 + int(s) for h, m, s in (t.split(':') for t in times))
    average_seconds = total_seconds // len(times)
    hours, remainder = divmod(average_seconds, 3600)
    minutes, seconds = divmod(remainder, 60)
    return f"{hours}:{minutes}:{seconds}"

# Calculate average time for each row
df['Average Convergency Time'] = df['Convergency time'].apply(average_time)

# Save the updated DataFrame to a new CSV file
df.to_csv('updated_data2.csv', index=False)

import pandas as pd

# Load the CSV file
df = pd.read_csv('input.csv')

# Function to calculate percentage difference
def percent_difference(row):
    time1 = pd.to_timedelta(row['Time1'])
    time2 = pd.to_timedelta(row['Time2'])
    if time1 == 0:
        return None
    return ((time2 - time1) / time1) * 100

# Apply the function to each row
df['Percent Difference'] = df.apply(percent_difference, axis=1)

# Save the output to a new CSV file
df.to_csv('output.csv', index=False)
