WebSocket Server

A minimal Node.js WebSocket server with broadcast capability

const WebSocket = require('ws'); const wss = new WebSocket.Server({ port: 8080 }); wss.on('connection', (ws) => { console.log('Client connected'); ws.on('message', (data) => { // Broadcast to all clients wss.clients.forEach((client) => { if (client.readyState === WebSocket.OPEN) { client.send(data.toString()); } }); }); ws.on('close', () => console.log('Client disconnected')); });
