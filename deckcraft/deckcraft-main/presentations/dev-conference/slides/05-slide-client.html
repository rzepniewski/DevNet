Browser-native WebSocket API with event handlers

const ws = new WebSocket('ws://localhost:8080'); ws.onopen = () => { console.log('Connected to server'); ws.send(JSON.stringify({ type: 'join', user: 'Alice' })); }; ws.onmessage = (event) => { const data = JSON.parse(event.data); renderMessage(data); }; ws.onclose = () => console.log('Disconnected'); ws.onerror = (err) => console.error('WebSocket error:', err);
